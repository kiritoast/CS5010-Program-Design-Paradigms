package problem2;

import java.util.Objects;

/**
 * Restaurant class that contain class address and menu
 */
public class Restaurant {
  private String name;
  private Address address;
  private Menu menu;
  private boolean openOrClosed;

  /**
   *
   * @param name name
   * @param address address
   * @param menu menu
   * @param openOrClosed boolean if the store is open or closed
   */
  public Restaurant(String name, Address address, Menu menu, boolean openOrClosed) {
    this.name = name;
    this.address = address;
    this.menu = menu;
    this.openOrClosed = openOrClosed;
  }

  /**
   * get name
   * @return name
   */
  public String getName() {
    return name;
  }

  /**
   * set the name
   * @param name name
   */
  public void setName(String name) {
    this.name = name;
  }

  /**
   * get the address
   * @return address
   */
  public Address getAddress() {
    return address;
  }

  /**
   * set address
   * @param address address
   */
  public void setAddress(Address address) {
    this.address = address;
  }

  /**
   * get menu
   * @return menu
   */
  public Menu getMenu() {
    return menu;
  }

  /**
   * set the menu
   * @param menu menu
   */
  public void setMenu(Menu menu) {
    this.menu = menu;
  }

  /**
   * get if the restaurants is open or closed
   * @return if the restaurants is open or closed
   */
  public boolean isOpenOrClosed() {
    return openOrClosed;
  }

  /**
   * set if the restaurants is open or closed
   * @param openOrClosed boolean
   */
  public void setOpenOrClosed(boolean openOrClosed) {
    this.openOrClosed = openOrClosed;
  }
  /**
   * {@inheritDoc}
   */
  @Override
  public boolean equals(Object o) {
    if (this == o) {
      return true;
    }
    if (o == null || getClass() != o.getClass()) {
      return false;
    }
    Restaurant that = (Restaurant) o;
    return openOrClosed == that.openOrClosed && Objects.equals(name, that.name)
        && Objects.equals(address, that.address) && Objects.equals(menu,
        that.menu);
  }
  /**
   * {@inheritDoc}
   */

  @Override
  public int hashCode() {
    return Objects.hash(name, address, menu, openOrClosed);
  }
  /**
   * {@inheritDoc}
   */

  @Override
  public String toString() {
    return "Restaurant{" +
        "name='" + name + '\'' +
        ", address=" + address +
        ", menu=" + menu +
        ", openOrClosed=" + openOrClosed +
        '}';
  }
}
