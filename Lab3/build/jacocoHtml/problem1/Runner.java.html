package problem1;

import java.util.Objects;

/**
 * runner class extends Athlete
 */
public class Runner extends Athlete{
  private double best5KTime;
  private double halfMarathon;
  private String favEvent;

  /**
   *
   * @param athletesName - object problem1.Name, containing athlete's first, middle and last name
   * @param height - athlete's height, expressed as a Double in cm (e.g., 6'2'' is recorded as 187.96cm)
   * @param weight - athlete's weigh, expressed as a Double in pounds (e.g. 125, 155, 200 pounds)
   * @param league - athelete's league, expressed as String
   * @param best5KTime best 5k time result
   * @param halfMarathon best half-marathon speed
   * @param favEvent favourite event
   */
  public Runner(Name athletesName, Double height, Double weight, String league, double best5KTime,
      double halfMarathon, String favEvent) {
    super(athletesName, height, weight, league);
    this.best5KTime = best5KTime;
    this.halfMarathon = halfMarathon;
    this.favEvent = favEvent;
  }

  /**
   *
   * @param athletesName - object problem1.Name, containing athlete's first, middle and last name
   * @param height - athlete's height, expressed as a Double in cm (e.g., 6'2'' is recorded as 187.96cm)
   * @param weight - athlete's weigh, expressed as a Double in pounds (e.g. 125, 155, 200 pounds)
   * @param best5KTime best 5k time result
   * @param halfMarathon best half-marathon speed
   * @param favEvent favourite event
   */
  public Runner(Name athletesName, Double height, Double weight, double best5KTime,
      double halfMarathon, String favEvent) {
    super(athletesName, height, weight);
    this.best5KTime = best5KTime;
    this.halfMarathon = halfMarathon;
    this.favEvent = favEvent;
  }

  /**
   *
   * @return best 5k time
   */
  public double getBest5KTime() {
    return best5KTime;
  }

  /**
   *
   * @return best half-marathon speed
   */
  public double getHalfMarathon() {
    return halfMarathon;
  }

  /**
   *
   * @return favourite event
   */
  public String getFavEvent() {
    return favEvent;
  }

  /**
   *
   * @param o object
   * @return true if same
   */
  @Override
  public boolean equals(Object o) {
    if (this == o) {
      return true;
    }
    if (o == null || getClass() != o.getClass()) {
      return false;
    }
    Runner runner = (Runner) o;
    return Double.compare(runner.best5KTime, best5KTime) == 0
        && Double.compare(runner.halfMarathon, halfMarathon) == 0
        && Objects.equals(favEvent, runner.favEvent);
  }

  /**
   *
   * @return hashcode of class
   */
  @Override
  public int hashCode() {
    return Objects.hash(best5KTime, halfMarathon, favEvent);
  }

  /**
   *
   * @return string of class
   */
  @Override
  public String toString() {
    return "problem1.Runner{" +
        "best5KTime=" + best5KTime +
        ", halfMarathon=" + halfMarathon +
        ", favEvent='" + favEvent + '\'' +
        '}';
  }
}
