package problem3;

/**
 * class contact info that store tax filer information
 */
public class ContactInfo {
  private Name name;
  private String address;
  private String phoneNumber;
  private String emailAddress;

  /**
   *
   * @param name string name
   * @param address string address
   * @param phoneNumber  the tax filer’s phone number
   * @param emailAddress the tax filer’s email address
   */
  public ContactInfo(Name name, String address, String phoneNumber, String emailAddress) {
    this.name = name;
    this.address = address;
    this.phoneNumber = phoneNumber;
    this.emailAddress = emailAddress;
  }

  /**
   *
   * @return the name of tax filer
   */
  public Name getName() {
    return name;
  }

  /**
   *
   * @return tax filer’s address
   */
  public String getAddress() {
    return address;
  }

  /**
   *
   * @return the tax filer’s phone number
   */
  public String getPhoneNumber() {
    return phoneNumber;
  }

  /**
   *
   * @return the tax filer’s email address
   */
  public String getEmailAddress() {
    return emailAddress;
  }

}
