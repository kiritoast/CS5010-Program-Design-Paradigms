package problem2;

import java.util.Objects;

/**
 * a custom class that keeps track of make and model
 */
public class MakeModel {
  private String Make;
  private String Model;

  /**
   *
   * @param make vehicle make, represented as a String
   * @param model vehicle model, represented as a String
   */
  public MakeModel(String make, String model) {
    Make = make;
    Model = model;
  }

  /**
   *
   * @return make
   */
  public String getMake() {
    return Make;
  }

  /**
   *
   * @return model
   */
  public String getModel() {
    return Model;
  }

  /**
   *
   * @param o  object and see if the same object
   * @return return true for same, and false for not
   */
  @Override
  public boolean equals(Object o) {
    if (this == o) {
      return true;
    }
    if (!(o instanceof MakeModel makeModel)) {
      return false;
    }
    return Objects.equals(getMake(), makeModel.getMake()) && Objects.equals(
        getModel(), makeModel.getModel());
  }

  /**
   *
   * @return the hashcode
   */
  @Override
  public int hashCode() {
    return Objects.hash(getMake(), getModel());
  }
}
