package problem2;

/**
 * abstract vehicle
 *
 */
public abstract class Vehicle {

  /**
   * string that store unique Vehicle id
   */
  protected String ID;
  /**
   * manufacturingYear year vehicle was manufactured
   */
  protected String manufacturingYear;
  /**
   * makeModel represented as custom class MakeModel
   */
  protected MakeModel makeModel;
  /**
   * MSRP Manufacturer Suggested Retail Price
   */
  protected double MSRP;

  /**
   *
   * @param ID unique identifier of a vehicle
   * @param manufacturingYear year vehicle was manufactured
   * @param makeModel represented as custom class MakeModel
   * @param MSRP Manufacturer Suggested Retail Price
   */
  public Vehicle(String ID, String manufacturingYear, MakeModel makeModel, double MSRP) {
    this.ID = ID;
    this.manufacturingYear = manufacturingYear;
    this.makeModel = makeModel;
    this.MSRP = MSRP;
  }

  /**
   *
   * @return unique id of vehicle
   */
  public String getID() {
    return ID;
  }

  /**
   *
   * @return year vehicle was manufactured
   */
  public String getManufacturingYear() {
    return manufacturingYear;
  }

  /**
   *
   * @return class MakeModel
   */
  public MakeModel getMakeModel() {
    return makeModel;
  }

  /**
   *
   * @return Manufacturer Suggested Retail Price
   */
  public double getMSRP() {
    return MSRP;
  }

}
