package problem1;

import java.util.HashMap;

public class TransferSystem {
  private HashMap <String, FrequentFlyer> map;

  public TransferSystem(HashMap<String, FrequentFlyer> map) {
    this.map = map;
  }
  public boolean checkFlyerExist(String accountID, FrequentFlyer frequentFlyer){
    if(map.containsKey(accountID)){
      return map.get(accountID).getFirstName().equals(frequentFlyer.getFirstName())
          && map.get(accountID).getMiddleName().equals(frequentFlyer.getMiddleName())
          && map.get(accountID).getLastName().equals(frequentFlyer.getLastName());
    }else{
      return false;
    }
  }
}
