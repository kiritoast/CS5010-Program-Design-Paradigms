package problem1;

import java.util.HashMap;

/**
 * class the check if the name is unique
 */
public class FlyerChecker {
  private HashMap <String, FrequentFlyer> map;

  /**
   *
   * @param map that stores the account id and the FrequentFlyer
   */
  public FlyerChecker(HashMap<String, FrequentFlyer> map) {
    this.map = map;
  }

  /**
   *
   * @param accountID unique account ID, which is a 12-character long String
   * @param frequentFlyer A frequent flyer class
   * @return true if he exists in the system and match the name, false if not
   */
  public boolean checkFlyerExist(String accountID, FrequentFlyer frequentFlyer){
    if(map.containsKey(accountID)){
      return map.get(accountID).getFirstName().equals(frequentFlyer.getFirstName())
          && map.get(accountID).getMiddleName().equals(frequentFlyer.getMiddleName())
          && map.get(accountID).getLastName().equals(frequentFlyer.getLastName());
    }else{
      return false;
    }
  }
}
