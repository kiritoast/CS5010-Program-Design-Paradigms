package problem2;

/**
 * subclass of car can see Mileage, numberOfPreviousOwner, numberOfPreviousAccident
 */
public class UsedCar extends Car {

  private int Mileage;
  private int numberOfPreviousOwner;
  private int numberOfPreviousAccident;

  /**
   *
   * @param ID unique identifier of a vehicle
   * @param manufacturingYear year vehicle was manufactured
   * @param makeModel represented as custom class MakeModel
   * @param MSRP Manufacturer Suggested Retail Price
   * @param mileage represented as an Integer
   * @param numberOfPreviousOwner Number of previous owners, represented as an Integer
   * @param numberOfPreviousAccident Number of minor traffic accidents the vehicle was involved, int
   */
  public UsedCar(String ID, String manufacturingYear, MakeModel makeModel, double MSRP, int mileage,
      int numberOfPreviousOwner, int numberOfPreviousAccident) {
    super(ID, manufacturingYear, makeModel, MSRP);
    Mileage = mileage;
    this.numberOfPreviousOwner = numberOfPreviousOwner;
    this.numberOfPreviousAccident = numberOfPreviousAccident;
  }

  /**
   *
   * @return the mileage the used car
   */
  public int getMileage() {
    return Mileage;
  }

  /**
   *
   * @return the number of previous owner
   */
  public int getNumberOfPreviousOwner() {
    return numberOfPreviousOwner;
  }

  /**
   *
   * @return the number of previous accident
   */
  public int getNumberOfPreviousAccident() {
    return numberOfPreviousAccident;
  }
}
