package problem2;

/**
 * class Boat that inherient Vessel
 */
public class Boat extends Vessel{

  private float length;
  private int numberOfPassenger;
  private PropulsionType type;

  /**
   *
   * @param ID  boat ID
   * @param manufacturingYear year of boat manufacture
   * @param makeModel class make model
   * @param MSRP Manufacturer Suggested Retail Price
   * @param length boat length
   * @param numberOfPassenger number of passenger
   * @param type represented as an PropulsionType with possible value: Sail Power, Inboard Engine, Outboard Engine, Jet Propulsion.
   */
  public Boat(String ID, String manufacturingYear, MakeModel makeModel, double MSRP, float length,
      int numberOfPassenger, PropulsionType type) {
    super(ID, manufacturingYear, makeModel, MSRP);
    this.length = length;
    this.numberOfPassenger = numberOfPassenger;
    this.type = type;
  }

  /**
   *
   * @return the length of the boat
   */
  public float getLength() {
    return length;
  }

  /**
   *
   * @return the number of passenger
   */
  public int getNumberOfPassenger() {
    return numberOfPassenger;
  }

  /**
   *
   * @return return the type of boat
   */
  public PropulsionType getType() {
    return type;
  }
}
