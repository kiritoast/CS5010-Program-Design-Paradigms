package problem1;

/**
 * class that contain total miles available, miles earned this year, miles expiring by
 * the end of this calendar year
 */
public class MileBalance {
  private int totalMiles;
  private int milesThisYear;
  private int milesExpiring;

  /**
   *
   * @param totalMiles total miles available
   * @param milesThisYear miles earned this year
   * @param milesExpiring miles expiring by the end of this calendar year
   */
  public MileBalance(int totalMiles, int milesThisYear, int milesExpiring) {
    this.totalMiles = totalMiles;
    this.milesThisYear = milesThisYear;
    this.milesExpiring = milesExpiring;
  }

  /**
   *
   * @return total miles available
   */
  public int getTotalMiles() {
    return totalMiles;
  }

  /**
   *
   * @return miles earned this year
   */
  public int getMilesThisYear() {
    return milesThisYear;
  }

  /**
   *
   * @return miles expiring by the end of this calendar year
   */
  public int getMilesExpiring() {
    return milesExpiring;
  }


  /**
   *
   * @param totalMiles set total miles to parameter totalMiles
   */
  public void setTotalMiles(int totalMiles) {
    this.totalMiles = totalMiles;
  }

  /**
   *
   * @param milesThisYear set miles earned this year to parameter milesThisYear
   */
  public void setMilesThisYear(int milesThisYear) {
    this.milesThisYear = milesThisYear;
  }

  /**
   *
   * @param milesExpiring set miles left this year to parameter milesExpiring
   */
  public void setMilesExpiring(int milesExpiring) {
    this.milesExpiring = milesExpiring;
  }
}
