package problem1;

import java.util.Objects;

/**
 * Deposit that allow transfer between account
 */
public class Deposit{
  private int amount;
  private FrequentFlyer user;
  private static final int LOWER_LIMIT = 1000;
  private static final int UPPER_LIMIT = 10000;

  /**
   *
   * @param amount Deposit amount that is in the range [1000 – 10000] miles
   * @param user information about the recipient
   * @throws Exception amount range between 1000 – 10000
   */
  public Deposit(int amount, FrequentFlyer user) throws Exception {
    if(amount > UPPER_LIMIT || amount < LOWER_LIMIT){
       throw new Exception("The mile should between 1000 and 10000 ");
    }
    this.amount = amount;
    this.user = user;
  }

  /**
   *
   * @return amount that transfer between class
   */
  public int getAmount() {
    return amount;
  }

  /**
   *
   * @return user who get the deposit
   */
  public FrequentFlyer getUser() {
    return user;
  }

  /**
   *
   * @param o object that see if eqauls
   * @return true if the same, false if not
   */
  @Override
  public boolean equals(Object o) {
    if (this == o) {
      return true;
    }
    if (o == null || getClass() != o.getClass()) {
      return false;
    }
    Deposit deposit = (Deposit) o;
    return amount == deposit.amount && Objects.equals(user, deposit.user);
  }

  /**
   *
   * @return hashcode of the object
   */
  @Override
  public int hashCode() {
    return Objects.hash(amount, user);
  }
}
