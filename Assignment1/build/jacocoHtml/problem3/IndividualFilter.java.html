package problem3;

/**
 * class individual filter that extends from tax filer
 */
public class IndividualFilter extends TaxFiler{

  /**
   *
   * @param taxID unique tax filer’s identifier
   * @param contactInfo class contact info that store tax filer information
   * @param lastYearEarning Last year’s earnings
   * @param paidIncomeTax Total income tax already paid
   * @param paidMortgageInterest Mortgage interest paid
   * @param paidPropertyTax Property taxes
   * @param paidStudentLoanTuition Student loan and tuition paid
   * @param contributionsRetirementSaving Contributions made to a retirement savings account
   * @param contributionHealthSaving Contributions made to a health savings account
   * @param charitableDonationContribution Charitable donations and contributions
   */
  public IndividualFilter(String taxID, ContactInfo contactInfo, double lastYearEarning,
      double paidIncomeTax, double paidMortgageInterest, double paidPropertyTax,
      double paidStudentLoanTuition, double contributionsRetirementSaving,
      double contributionHealthSaving, double charitableDonationContribution) {
    super(taxID, contactInfo, lastYearEarning, paidIncomeTax, paidMortgageInterest, paidPropertyTax,
        paidStudentLoanTuition, contributionsRetirementSaving, contributionHealthSaving,
        charitableDonationContribution);
  }

  /**
   *
   * @return calculate the Retirement and health savings deduction
   */
  @Override
  protected Double calculateRetirementAndHealthSavingsDeduction() {
    double deduction = (contributionsRetirementSaving + contributionHealthSaving) * 0.7;
    if (deduction < calculateBasicTaxableIncome()){
      return deduction;
    }else{
      return calculateBasicTaxableIncome();
    }
  }

  /**
   *
   * @return the tax amount is calculated by taking the resulting taxable income
   */
  @Override
  public Double calculateTaxes() {
    double taxableIncome = calculateBasicTaxableIncome() - calculateRetirementAndHealthSavingsDeduction() -
        calculateMortgageInterestAndPropertyDeduction();
    if(taxableIncome < 55000){
      return taxableIncome * 0.15;
    } else{
      return taxableIncome * 0.19;
    }
  }
}
