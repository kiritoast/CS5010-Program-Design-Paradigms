package problem3;

/**
 * abstract class for all tax filer
 */
public abstract class TaxFiler {

  /**
   * string that store unique tax id
   */
  protected String taxID;
  /**
   * class that store all user contact info
   */
  protected ContactInfo contactInfo;
  /**
   * double that stores last year earning
   */
  protected double lastYearEarning;
  /**
   * double that store paid income tax
   */
  protected double paidIncomeTax;
  /**
   * double that store paid mortgage interest
   */
  protected double paidMortgageInterest;
  /**
   * double that store paid property tax
   */
  protected double paidPropertyTax;
  /**
   * double that store paid student loan
   */
  protected double paidStudentLoanTuition;
  /**
   * number of contribution to retirement saving
   */
  protected double contributionsRetirementSaving;
  /**
   * number of contribution to health saving
   */
  protected double contributionHealthSaving;
  /**
   * charitableDonationContribution Charitable donations and contributions
   */
  protected double charitableDonationContribution;

  /**
   *
   * @param taxID unique tax filer’s identifier
   * @param contactInfo class contact info that store tax filer information
   * @param lastYearEarning Last year’s earnings
   * @param paidIncomeTax Total income tax already paid
   * @param paidMortgageInterest Mortgage interest paid
   * @param paidPropertyTax Property taxes
   * @param paidStudentLoanTuition Student loan and tuition paid
   * @param contributionsRetirementSaving Contributions made to a retirement savings account
   * @param contributionHealthSaving Contributions made to a health savings account
   * @param charitableDonationContribution Charitable donations and contributions
   */
  public TaxFiler(String taxID, ContactInfo contactInfo, double lastYearEarning,
      double paidIncomeTax,
      double paidMortgageInterest, double paidPropertyTax, double paidStudentLoanTuition,
      double contributionsRetirementSaving, double contributionHealthSaving,
      double charitableDonationContribution) {
    this.taxID = taxID;
    this.contactInfo = contactInfo;
    this.lastYearEarning = lastYearEarning;
    this.paidIncomeTax = paidIncomeTax;
    this.paidMortgageInterest = paidMortgageInterest;
    this.paidPropertyTax = paidPropertyTax;
    this.paidStudentLoanTuition = paidStudentLoanTuition;
    this.contributionsRetirementSaving = contributionsRetirementSaving;
    this.contributionHealthSaving = contributionHealthSaving;
    this.charitableDonationContribution = charitableDonationContribution;
  }

  /**
   *
   * @return Basic taxable income for all tax filer
   */
  protected Double calculateBasicTaxableIncome() {
    return lastYearEarning - paidIncomeTax;
  }

  /**
   *
   * @return abstract method Retirement and health savings deduction for all tax filer
   */
  protected abstract Double calculateRetirementAndHealthSavingsDeduction();

  /**
   *
   * @return Mortgage interest and property deduction
   */
  protected Double calculateMortgageInterestAndPropertyDeduction(){
    if(lastYearEarning < 250000.0 && (paidMortgageInterest + paidPropertyTax) > 12500.0){
      return 2500.0;
    }
    return 0.00;
  }

  /**
   *
   * @return abstract that calculate Tax amount for all
   */
  protected abstract Double calculateTaxes();
}
