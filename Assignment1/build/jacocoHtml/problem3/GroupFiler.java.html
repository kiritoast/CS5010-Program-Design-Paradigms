package problem3;

/**
 * group filer that extends from tax filer
 */
public class GroupFiler extends TaxFiler{

  /**
   * numberDependents Number of dependents
   */
  public int numberDependents;
  /**
   * numberMinorChildren Number of minor children
   */
  public int numberMinorChildren;
  /**
   * childcareExpenses Childcare expenses
   */
  public double childcareExpenses;
  /**
   * dependentCareExpenses Dependent-care expenses
   */
  public double dependentCareExpenses;

  /**
   *
   * @param taxID unique tax filer’s identifier
   * @param contactInfo class contact info that store tax filer information
   * @param lastYearEarning Last year’s earnings
   * @param paidIncomeTax Total income tax already paid
   * @param paidMortgageInterest Mortgage interest paid
   * @param paidPropertyTax Property taxes
   * @param paidStudentLoanTuition Student loan and tuition paid
   * @param contributionsRetirementSaving Contributions made to a retirement savings account
   * @param contributionHealthSaving Contributions made to a health savings account
   * @param charitableDonationContribution Charitable donations and contributions
   * @param numberDependents Number of dependents
   * @param numberMinorChildren Number of minor children
   * @param childcareExpenses Childcare expenses
   * @param dependentCareExpenses Dependent-care expenses
   */
  public GroupFiler(String taxID, ContactInfo contactInfo, double lastYearEarning,
      double paidIncomeTax, double paidMortgageInterest, double paidPropertyTax,
      double paidStudentLoanTuition, double contributionsRetirementSaving,
      double contributionHealthSaving, double charitableDonationContribution, int numberDependents,
      int numberMinorChildren, double childcareExpenses, double dependentCareExpenses) {
    super(taxID, contactInfo, lastYearEarning, paidIncomeTax, paidMortgageInterest, paidPropertyTax,
        paidStudentLoanTuition, contributionsRetirementSaving, contributionHealthSaving,
        charitableDonationContribution);
    this.numberDependents = numberDependents;
    this.numberMinorChildren = numberMinorChildren;
    this.childcareExpenses = childcareExpenses;
    this.dependentCareExpenses = dependentCareExpenses;
  }

  /**
   *
   * @return Number of dependents
   */
  public int getNumberDependents() {
    return numberDependents;
  }

  /**
   *
   * @return Number of minor children
   */
  public int getNumberMinorChildren() {
    return numberMinorChildren;
  }

  /**
   *
   * @return Childcare expenses
   */
  public double getChildcareExpenses() {
    return childcareExpenses;
  }

  /**
   *
   * @return Dependent-care expenses
   */
  public double getDependentCareExpenses() {
    return dependentCareExpenses;
  }

  /**
   *
   * @return Retirement and health savings deduction
   */
  @Override
  protected Double calculateRetirementAndHealthSavingsDeduction() {
    double deduction = (contributionsRetirementSaving + contributionHealthSaving) * 0.65;
    if (deduction < calculateBasicTaxableIncome()){
      if(deduction > 17500.00){
        return 17500.00;
      }else{
        return deduction;
      }
    }else{
      return calculateBasicTaxableIncome();
    }

  }

  /**
   *
   * @return the tax
   */
  @Override
  public Double calculateTaxes() {
    double taxableIncome = calculateBasicTaxableIncome()-calculateChildrenDeduction() -
        calculateMortgageInterestAndPropertyDeduction();
    if(taxableIncome < 90000 ){
      return taxableIncome * 0.145;
    }else{
      return taxableIncome * 0.185;
    }
  }

  /**
   *
   * @return Childcare deduction
   */
  protected Double calculateChildrenDeduction(){
    if(calculateBasicTaxableIncome() < 200000 && childcareExpenses > 5000){
      return 1250.00;
    }
    return 0.00;
  }
}
