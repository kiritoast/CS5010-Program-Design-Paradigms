package problem2;

/**
 * abstract type for boat that extends vehicle
 */
public abstract class Vessel extends Vehicle{

  /**
   *
   * @param ID unique identifier of a vehicle
   * @param manufacturingYear year vehicle was manufactured
   * @param makeModel represented as custom class MakeModel
   * @param MSRP Manufacturer Suggested Retail Price
   */
  public Vessel(String ID, String manufacturingYear, MakeModel makeModel, double MSRP) {
    super(ID, manufacturingYear, makeModel, MSRP);
  }
}
